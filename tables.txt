CREATE TABLE users(email varchar(50) primary key,mobile varchar(50) UNIQUE)

CREATE TABLE emergency(id varchar(50),date datetime DEFAULT(getdate()),foreign key(id) references users(email))

CREATE TABLE daily(id varchar(50),score int,date datetime DEFAULT(getdate()),foreign key(id) references users(email))

CREATE TABLE prescription(id varchar(50),morning varchar(50),afternoon varchar(50),evening varchar(50),foreign key(id) references users(email))
